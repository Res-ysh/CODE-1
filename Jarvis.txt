# ─────────────────────────────────────────
#  VOICE ENGINE
# ─────────────────────────────────────────


recognizer = sr.Recognizer()
recognizer.pause_threshold = 0.8

# ─────────────────────────────────────────
#  SPEAK
# ─────────────────────────────────────────
def speak(text):
    print(f"\nJARVIS: {text}\n")
    engine = pyttsx3.init()
    voices = engine.getProperty("voices")
    engine.setProperty("voice", voices[0].id)
    engine.setProperty("rate", 220)
    engine.setProperty("volume", 1.0)
    engine.say(text)
    engine.runAndWait()
    engine.stop()
# ─────────────────────────────────────────
#  LISTEN
# ─────────────────────────────────────────
def listen():
    with sr.Microphone() as source:
        print("[Listening... speak now]")
        recognizer.adjust_for_ambient_noise(source, duration=0.3)
        try:
            audio = recognizer.listen(source, timeout=7, phrase_time_limit=10)
            query = recognizer.recognize_google(audio).lower()
            print(f"You: {query}")
            return query
        except sr.WaitTimeoutError:
            return ""
        except sr.UnknownValueError:
            return ""
        except sr.RequestError:
            speak("I'm having trouble with the speech service. Check your internet.")
            return ""

# ─────────────────────────────────────────
#  AI BRAIN
# ─────────────────────────────────────────
def ask_groq(user_input):
    conversation_history.append({"role": "user", "content": user_input})
    try:
        response = client.chat.completions.create(
            model=MODEL,
            messages=conversation_history,
            max_tokens=200,
            temperature=0.85,
        )
        reply = response.choices[0].message.content.strip()
        conversation_history.append({"role": "assistant", "content": reply})
        return reply
    except Exception as e:
        return f"My brain glitched for a second. Error: {str(e)}"

# ─────────────────────────────────────────
#  GREET
# ─────────────────────────────────────────
def greet():
    hour = datetime.datetime.now().hour
    if hour < 12:
        period = "morning"
    elif hour < 17:
        period = "afternoon"
    else:
        period = "evening"
    speak(f"Good {period}, {NAME}. JARVIS online. What's on your mind?")

# ─────────────────────────────────────────
#  QUICK ACTIONS
# ─────────────────────────────────────────
def try_quick_action(query):
    if any(w in query for w in ["exit", "quit", "goodbye", "bye", "shut down yourself"]):
        speak(f"Alright, catch you later {NAME}. Don't cause too much trouble.")
        sys.exit()

    if "time" in query and "what" in query:
        now = datetime.datetime.now().strftime("%I:%M %p")
        speak(f"It's {now}.")
        return True

    if "date" in query or "what day" in query:
        today = datetime.datetime.now().strftime("%A, %B %d, %Y")
        speak(f"Today is {today}.")
        return True

    if "open" in query:
        for name, path in APPS.items():
            if name in query:
                speak(f"Opening {name}.")
                try:
                    subprocess.Popen(path)
                except FileNotFoundError:
                    speak(f"I can't find {name} at the usual path. Update the path in the config.")
                return True

    for name, url in WEBSITES.items():
        if name in query and any(w in query for w in ["open", "go to", "show"]):
            speak(f"Opening {name}.")
            webbrowser.open(url)
            return True

    for kw in ["search for", "google", "look up", "search"]:
        if kw in query:
            term = query.replace(kw, "").strip()
            if term:
                speak(f"Searching for {term}.")
                webbrowser.open(f"https://www.google.com/search?q={term.replace(' ', '+')}")
                return True

    try:
        import pyautogui
        if "volume up" in query:
            for _ in range(5): pyautogui.press("volumeup")
            speak("Done.")
            return True
        if "volume down" in query:
            for _ in range(5): pyautogui.press("volumedown")
            speak("Done.")
            return True
        if "mute" in query:
            pyautogui.press("volumemute")
            speak("Muted.")
            return True
    except ImportError:
        pass

    if "lock" in query and "screen" in query:
        speak("Locking the screen.")
        os.system("rundll32.exe user32.dll,LockWorkStation")
        return True
    if "shutdown" in query:
        speak("Shutting down in 5 seconds.")
        os.system("shutdown /s /t 5")
        return True
    if "restart" in query:
        speak("Restarting now.")
        os.system("shutdown /r /t 5")
        return True

    return False

# ─────────────────────────────────────────
#  MAIN LOOP
# ─────────────────────────────────────────
def main():
    greet()
    while True:
        query = listen()
        if not query:
            continue
        if try_quick_action(query):
            continue
        print("[Thinking...]")
        reply = ask_groq(query)
        speak(reply)

if __name__ == "__main__":
    main()
